\begin{abstract}
This companion document contains the \textbf{complete set of 36 technical appendices} (A through AJ) to ``Complementarity and Context: A Unified Framework for Economic Rationality.''

\textbf{Part I: Core Technical Material (A--G)}
\begin{itemize}[noitemsep]
    \item Appendix A: Formal Derivations of Limiting Cases
    \item Appendix B: Complementarity Conditions by Level
    \item Appendix C: The Complete 144-Component FEPSDE Matrix
    \item Appendix D: Mathematical Proofs
    \item Appendix E: Empirical Operationalization
    \item Appendix F: Worked Examples Across Levels
    \item Appendix G: Glossary and Notation
\end{itemize}

\textbf{Part II: Computational and Historical Context (H--J)}
\begin{itemize}[noitemsep]
    \item Appendix H: Computational History of Complementarity Analysis
    \item Appendix I: Nobel Contributions as Framework Components (1969--2024)
    \item Appendix J: Prospective Applications (2024--2025)
\end{itemize}

\textbf{Part III: Research Tradition Surveys (K--Q)}
\begin{itemize}[noitemsep]
    \item Appendix K: The Zurich Tradition (Ernst Fehr)
    \item Appendix L: Institutional Economics (Daron Acemoglu)
    \item Appendix M: Financial Economics (Andrei Shleifer)
    \item Appendix N: Human Capital (James Heckman)
    \item Appendix O: Labor Economics (David Autor)
    \item Appendix P: Development Economics (Esther Duflo)
    \item Appendix Q: Uncertainty and Management (Nicholas Bloom)
\end{itemize}

\textbf{Part IV: Framework Validation (R--T)}
\begin{itemize}[noitemsep]
    \item Appendix R: Framework Evaluation Protocol
    \item Appendix S: Falsifiable Predictions
    \item Appendix T: The Framework as Metatheory
\end{itemize}

\textbf{Part V: Extended Theoretical Foundations (U--Z)}
\begin{itemize}[noitemsep]
    \item Appendix U: Kahneman and Tversky
    \item Appendix V: The Eight $\Psi$ Dimensions
    \item Appendix W: Information Economics
    \item Appendix X: Milgrom and Roberts
    \item Appendix Y: Capital Markets
    \item Appendix Z: Growth Theory
\end{itemize}

\textbf{Part VI: Domain Applications (AA--AJ)}
\begin{itemize}[noitemsep]
    \item Appendix AA: Labor Economics Applications
    \item Appendix AB: Matching and Repugnant Markets
    \item Appendix AC: Industrial Organization
    \item Appendix AD: Evolutionary and Behavioral Game Theory
    \item Appendix AE: Mechanism Design
    \item Appendix AF: Social Choice Theory
    \item Appendix AG: Complexity Economics
    \item Appendix AH: Temporal Context
    \item Appendix AI: Spatial Context
    \item Appendix AJ: Social Preferences
\end{itemize}

\vspace{0.5em}
\noindent\textbf{Note:} This document is designed to be read alongside the main text. Each appendix can also stand alone as a reference document.
\end{abstract}

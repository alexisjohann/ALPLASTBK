\section{Abstract}
% =============================================================================

Claudia Goldin's Nobel Prize-winning research has fundamentally shaped our
understanding of \textbf{gender and the economy}. Through meticulous historical
analysis and innovative empirical methods, she has documented the long-run
evolution of women's labor force participation, education, and earnings---revealing
how social norms, technology, and institutional constraints interact to shape
economic outcomes.

This appendix integrates 30 papers spanning 1982--2023, organized around six
core axioms that connect Goldin's findings to EBF's complementarity-context
framework.

\section{Fundamental Question}
% =============================================================================

\begin{tcolorbox}[colback=red!5!white, colframe=red!75!black,
    title=The Goldin Question]
\centering\Large
\textbf{Why has the gender gap in earnings persisted despite the grand
convergence in education and labor force participation?}
\end{tcolorbox}

Goldin's answer: The remaining gap is largely explained by \textit{temporal
flexibility}---women disproportionately bear caregiving responsibilities,
leading them to sort into occupations with lower penalties for flexible hours.
``Greedy jobs'' with high nonlinear returns to long hours create persistent
gaps even absent discrimination.

\section{The Six Goldin Axioms}
% =============================================================================

\subsection{Axiom UNMAPPED_CG-1: The Grand Gender Convergence}

\begin{tcolorbox}[colback=blue!5!white, colframe=blue!75!black]
\textbf{Axiom UNMAPPED_CG-1 (Convergence Pattern):}
\textit{Over the 20th century, gender gaps in education, labor force
participation, and occupational distribution have substantially narrowed,
constituting a ``grand convergence'' that is largely complete except for
earnings.}
\end{tcolorbox}

\textbf{Key Papers:}
\begin{itemize}[nosep]
    \item Goldin (2014): ``A Grand Gender Convergence: Its Last Chapter''
    \item Goldin (2006): ``The Quiet Revolution''
    \item Goldin (1990): ``Understanding the Gender Gap''
\end{itemize}

\textbf{EBF Integration:}
The convergence represents a shift in context $\Psi$ over time:
\begin{equation}
\Psi_{gender}(t) \rightarrow \Psi_{neutral} \quad \text{as } t \rightarrow \text{present}
\end{equation}

However, residual $\Psi_{family}$ constraints persist, explaining remaining gaps.

\subsection{Axiom UNMAPPED_CG-2: Greedy Jobs and Nonlinear Returns}

\begin{tcolorbox}[colback=blue!5!white, colframe=blue!75!black]
\textbf{Axiom UNMAPPED_CG-2 (Greedy Jobs):}
\textit{Many high-paying occupations exhibit nonlinear returns to hours:
earnings increase more than proportionally with hours worked. This creates
a penalty for flexibility that disproportionately affects women with
caregiving responsibilities.}
\end{tcolorbox}

\textbf{Key Papers:}
\begin{itemize}[nosep]
    \item Goldin (2021): ``Career and Family''
    \item Goldin \& Katz (2011): ``The Cost of Workplace Flexibility''
    \item Goldin (2015): ``Hours Flexibility and the Gender Gap in Pay''
\end{itemize}

\textbf{EBF Integration:}
Greedy jobs create complementarity between hours and returns:
\begin{equation}
W(h) = w_0 \cdot h^{1+\gamma} \quad \text{where } \gamma > 0 \text{ (greedy job)}
\end{equation}

The $\gamma > 0$ parameter reflects the hours-compensation complementarity that
drives gender gaps.

\subsection{Axiom UNMAPPED_CG-3: The Pill and Career Investment}

\begin{tcolorbox}[colback=blue!5!white, colframe=blue!75!black]
\textbf{Axiom UNMAPPED_CG-3 (Contraception Effect):}
\textit{Access to reliable contraception (``the pill'') enabled women to
delay marriage and childbearing, allowing greater investment in education
and career---a key driver of the quiet revolution.}
\end{tcolorbox}

\textbf{Key Papers:}
\begin{itemize}[nosep]
    \item Goldin \& Katz (2002): ``The Power of the Pill''
\end{itemize}

\textbf{EBF Integration:}
The pill changed the expected returns to career investment:
\begin{equation}
E[U^{career}_{pre-pill}] < E[U^{career}_{post-pill}]
\end{equation}

By reducing $P(unplanned\_birth)$, contraception increased the expected payoff
to human capital investment for women.

\subsection{Axiom UNMAPPED_CG-4: U-Shaped Female Labor Participation}

\begin{tcolorbox}[colback=blue!5!white, colframe=blue!75!black]
\textbf{Axiom UNMAPPED_CG-4 (Development Pattern):}
\textit{Female labor force participation follows a U-shape across economic
development: high in agricultural societies, declining in early
industrialization (stigma + rising male wages), then rising again with
education and service sector growth.}
\end{tcolorbox}

\textbf{Key Papers:}
\begin{itemize}[nosep]
    \item Goldin (1995): ``The U-Shaped Female Labor Force Function''
\end{itemize}

\textbf{EBF Integration:}
The U-shape reflects changing context effects:
\begin{equation}
\alpha^{female}_{work}(\Psi_{development}) = \begin{cases}
\text{high} & \text{agricultural} \\
\text{low} & \text{early industrial (stigma)} \\
\text{high} & \text{post-industrial}
\end{cases}
\end{equation}

Social norms ($\Psi_{stigma}$) suppress participation even when economically
rational.

\subsection{Axiom UNMAPPED_CG-5: Blind Auditions and Hidden Discrimination}

\begin{tcolorbox}[colback=blue!5!white, colframe=blue!75!black]
\textbf{Axiom UNMAPPED_CG-5 (Discrimination Detection):}
\textit{Clever research designs can reveal discrimination that is otherwise
difficult to detect. Blind auditions in orchestras increased female hiring
by 30--55\%, demonstrating substantial evaluator bias.}
\end{tcolorbox}

\textbf{Key Papers:}
\begin{itemize}[nosep]
    \item Goldin \& Rouse (2000): ``Orchestrating Impartiality''
\end{itemize}

\textbf{EBF Integration:}
Evaluator bias affects the awareness function:
\begin{equation}
A_{evaluator}(quality | female) < A_{evaluator}(quality | male)
\end{equation}

Blind evaluation removes $\Psi_{gender\_visible}$, equalizing awareness of
quality.

\subsection{Axiom UNMAPPED_CG-6: Technology-Skill Complementarity}

\begin{tcolorbox}[colback=blue!5!white, colframe=blue!75!black]
\textbf{Axiom UNMAPPED_CG-6 (Education-Technology Race):}
\textit{The wage structure is shaped by a ``race'' between technological
change (increasing demand for skill) and educational attainment (increasing
supply). Periods when technology outpaces education see rising inequality.}
\end{tcolorbox}

\textbf{Key Papers:}
\begin{itemize}[nosep]
    \item Goldin \& Katz (2008): ``The Race between Education and Technology''
    \item Goldin \& Katz (1998): ``Origins of Technology-Skill Complementarity''
\end{itemize}

\textbf{EBF Integration:}
Technology-skill complementarity ($\gamma_{tech,skill} > 0$) means:
\begin{equation}
\frac{\partial^2 Y}{\partial Tech \partial Skill} > 0
\end{equation}

This is a direct application of EBF's CORE-HOW (B) complementarity framework.

\subsection{The Greedy Jobs Model in EBF Terms}

Goldin's ``greedy jobs'' concept maps directly to EBF complementarity:

\begin{equation}
U^{career} = w(h, \gamma) + U^{family}(T-h)
\end{equation}

where:
\begin{itemize}[nosep]
    \item $w(h, \gamma) = w_0 \cdot h^{1+\gamma}$ with $\gamma > 0$ for greedy jobs
    \item $T-h$ = time for family/caregiving
    \item Optimal $h^*$ depends on $\gamma$ and family constraints
\end{itemize}

Women with caregiving constraints choose lower $h$, accepting
disproportionately lower earnings when $\gamma > 0$.

\section{Worked Example: Pharmacy vs. Law}
% =============================================================================

\begin{tcolorbox}[colback=green!5!white, colframe=green!75!black,
    title=Example: Why Pharmacy Has No Gender Gap]

\textbf{Goldin \& Katz (2016) Finding:}
Pharmacy has essentially eliminated its gender earnings gap, while law
maintains a substantial gap.

\textbf{Explanation via UNMAPPED_CG-2 (Greedy Jobs):}
\begin{itemize}
    \item \textbf{Pharmacy:} Corporate chains, standardized protocols, perfect
    substitutability among pharmacists $\Rightarrow \gamma \approx 0$
    \item \textbf{Law:} Client relationships, partner track, long hours expected
    $\Rightarrow \gamma > 0$
\end{itemize}

\textbf{EBF Model:}
\begin{align}
W_{pharmacy}(h) &= w_0 \cdot h^{1.0} \quad (\gamma = 0) \\
W_{law}(h) &= w_0 \cdot h^{1.3} \quad (\gamma = 0.3)
\end{align}

\textbf{Policy Implication:}
Reducing $\gamma$ through technology, standardization, and team-based work
can close gender gaps without requiring women to work longer hours.

\end{tcolorbox}

\section{Critical Foundations}
% =============================================================================

\begin{enumerate}
    \item \textbf{Historical Method:} Goldin pioneered using economic history
    to understand contemporary issues---long time series reveal patterns
    invisible in cross-sections.

    \item \textbf{Natural Experiments:} Clever identification (blind auditions,
    pill access variation) provides causal evidence.

    \item \textbf{Complementarity Link:} Greedy jobs / technology-skill
    complementarity directly connect to EBF's CORE-HOW.

    \item \textbf{Context Effects:} Historical norms ($\Psi_{stigma}$, marriage
    bars) demonstrate context modulation of behavior.

    \item \textbf{Policy Relevance:} Direct implications for workplace design,
    parental leave, and anti-discrimination policy.
\end{enumerate}

\section{P\_eff Master Formula}
\label{sec:calc-peff}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\subsection{The Complete P\_eff Specification}

The probability of effective behavior change is computed as:

\begin{equation}
\boxed{
P_{eff}(t) = A_d(t) \cdot W_{AX,d}(s, \Psi) \cdot \kappa_{JNY}(\phi_d) \cdot \kappa_{KON}(c) \cdot \kappa_{AWX}(f)
}
\label{eq:peff-complete}
\end{equation}

where:
\begin{itemize}
\item $A_d(t)$ = Domain-specific awareness at time $t$
\item $W_{AX,d}(s, \Psi)$ = Willingness given signal $s$ and context $\Psi$
\item $\kappa_{JNY}(\phi_d)$ = Journey phase multiplier for phase $\phi$
\item $\kappa_{KON}(c)$ = Contextual factor for choice architecture
\item $\kappa_{AWX}(f)$ = Awareness-willingness coupling with feedback $f$
\end{itemize}

\section{Awareness Dynamics (Domain-Specific)}
\label{sec:calc-awareness}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\subsection{The Awareness ODE}

Awareness evolves according to domain-specific dynamics (from Appendix UNMAPPED_OOO):

\begin{equation}
\boxed{
\frac{dA_d}{dt} = \alpha_d \cdot T \cdot (1 - A_d) - \beta_d \cdot A_d
}
\label{eq:awareness-ode}
\end{equation}

where:
\begin{itemize}
\item $\alpha_d$ = Learning rate for domain $d$
\item $\beta_d$ = Decay rate for domain $d$
\item $T$ = Treatment intensity (0-1)
\end{itemize}

\subsection{Equilibrium Solution}

At equilibrium ($\frac{dA_d}{dt} = 0$):

\begin{equation}
A_d^* = \frac{\alpha_d \cdot T}{\alpha_d \cdot T + \beta_d}
\label{eq:awareness-equilibrium}
\end{equation}

\subsection{Convergence Time}

Time to reach 95\% of equilibrium:

\begin{equation}
t_{95} = \frac{3}{\alpha_d \cdot T + \beta_d}
\label{eq:convergence-time}
\end{equation}

\section{Stage Dynamics (φ Evolution)}
\label{sec:calc-stage}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\subsection{The BCJ Phase Evolution}

Stage (phase) evolves according to (from Appendix UNMAPPED_AW):

\begin{equation}
\boxed{
\frac{d\phi_d}{dt} = \mu_d(\phi_d) \cdot [1 + \eta_d \cdot I_d(t)]
}
\label{eq:phase-ode}
\end{equation}

where:
\begin{itemize}
\item $\mu_d(\phi_d)$ = Natural progression rate (phase-dependent)
\item $\eta_d$ = Intervention sensitivity
\item $I_d(t)$ = Intervention intensity at time $t$
\end{itemize}

\subsection{Phase-Type Affinity ($\alpha_{eff}$)}

The effective learning rate is modified by phase-type affinity:

\begin{equation}
\alpha_{eff} = \alpha_d \cdot \alpha_{PT}(\phi, T_j) \cdot \sigma_{ST}(S, T_j)
\label{eq:alpha-eff}
\end{equation}

where:
\begin{itemize}
\item $\alpha_{PT}(\phi, T_j)$ = Phase-Type Affinity Matrix value (Chapter 18)
\item $\sigma_{ST}(S, T_j)$ = Segment-Type Multiplier (Chapter 19)
\end{itemize}

\subsection{The Spillover Formula}

From Appendix UNMAPPED_COR, child initialization is computed as:

\begin{equation}
\boxed{
\phi_0^{child}(d') = \phi_{base}(d') + \sum_d \gamma^{inter}(d, d') \cdot (\phi^{parent}(d) - \bar{\phi})
}
\label{eq:spillover}
\end{equation}

where:
\begin{itemize}
\item $\gamma^{inter}(d, d')$ = Spillover coefficient from parent domain $d$ to child domain $d'$
\item $\bar{\phi}$ = Population mean state

\begin{equation}
\boxed{

where:
\begin{itemize}

\section{Portfolio Optimization}
\label{sec:calc-portfolio}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\subsection{The Optimization Problem}

Maximize expected behavior change subject to budget constraint:

\begin{equation}
\max_{I_1, ..., I_9} \quad P_{eff}(I_1, ..., I_9) \quad \text{s.t.} \quad \sum_{j} c_j \cdot I_j \leq B
\label{eq:portfolio-optimization}
\end{equation}

where $c_j$ is the cost per unit of intervention dimension $j$ (10C) and $B$ is the total budget.

\subsection{Gradient for Optimization}

The marginal effect of adding intervention type $j$:

\begin{equation}
\frac{\partial P_{eff}}{\partial T_j} = \gamma_j + \sum_{k \neq j} \gamma_{jk} \cdot T_k
\label{eq:portfolio-gradient}
\end{equation}

\subsection{Optimal Allocation Rule}

Allocate to type $j^*$ where:

\begin{equation}
j^* = \arg\max_j \frac{\gamma_j + \sum_{k \neq j} \gamma_{jk} \cdot T_k}{c_j}
\label{eq:optimal-allocation}
\end{equation}

\section{Worked Examples}
\label{sec:calc-examples}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\subsection{Example 1: Health Behavior Change}

\paragraph{Setup.} Design an intervention to increase physical activity in 35-year-olds.

\paragraph{Parameters.}
\begin{itemize}
\item Domain: Health → $\alpha = 0.05$, $\beta = 0.02$
\item Age decade: 30s → Phase: ``action'', Optimal dimensions: READY, WHEN, WHAT$_S$
\item Intervention: WHAT$_S$ (Social, peer exercise) with READY (Feedback, step tracking)
\end{itemize}

\paragraph{Calculation.}
\begin{align}
\alpha_{eff} &= 0.05 \times 1.4 \times 1.0 = 0.07 \quad \text{(WHAT$_S$ affinity)} \\
A^*_{health} &= \frac{0.07 \times 0.8}{0.07 \times 0.8 + 0.02} = 0.74 \\
t_{95} &= \frac{3}{0.07 \times 0.8 + 0.02} = 38 \text{ weeks} \\
P_{eff} &= 0.74 \times 0.6 \times 1.2 \times 1.1 = 0.59
\end{align}

\paragraph{Result.} Expected 59\% probability of sustained behavior change after 38 weeks.

\subsection{Example 2: Intergenerational Financial Planning}

\paragraph{Setup.} Family-level intervention targeting parental self-governance to improve child's money domain.

\paragraph{Parameters.}
\begin{itemize}
\item Parent domain: Self-Governance → $\alpha = 0.10$, fast learning
\item Child domain: Money → initialization offset
\item Spillover: $\gamma^{inter}(\text{Self-Gov}, \text{Money}) = 0.18$
\item Child age: 16 (within CTW 12--22)
\end{itemize}

\paragraph{Calculation.}
\begin{align}
\phi_0^{child}(\text{Money}) &= \phi_{base} + 0.18 \times (\phi^{parent}_{SG} - \bar{\phi}) \\
&= 0.3 + 0.18 \times (0.8 - 0.5) = 0.354 \\
ROI_{family} &= 1.7 \times ROI_{individual}
\end{align}

\paragraph{Result.} 18\% transmission of parental self-governance improvement to child's money initialization, with 70\% ROI amplification.

\section{Implications for Practice}
\label{sec:calc-practice}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{tcolorbox}[colback=green!5!white, colframe=green!75!black,
    title=Practical Implications]
\begin{enumerate}
\item \textbf{Domain Selection:} Choose domains with high $\alpha$ for quick wins (Self-Governance, Health) or high spillover for leverage (Self-Governance → multiple child domains).

\item \textbf{Dimension Selection:} Match intervention dimension to domain affinity. Avoid WHAT$_F$ (Financial) in Meaning domain ($\alpha_{eff} = 0.3$, backfire risk).

\item \textbf{Timing:} Target critical transmission windows for intergenerational interventions. Self-governance: ages 3--12; Money: ages 12--22.

\item \textbf{Portfolio Design:} Avoid WHAT$_S$+WHAT$_F$ and WHAT$_F$+HOW combinations due to crowding-out ($\gamma_{jk} < 0$).

\item \textbf{ROI Maximization:} Consider family-level planning for 70\% ROI amplification.
\end{enumerate}
\end{tcolorbox}

\section{Critical Foundations and Objections}
\label{sec:method-calc-foundations}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\subsection{Objection 1: Scope Limitations}

\textbf{Objection:} The framework presented may have limited applicability
outside the specific domain contexts discussed.

\textbf{Response:} We acknowledge domain-specificity. The framework provides
general principles that should be calibrated to specific contexts. Cross-domain
validation remains an open research question.

\subsection{Objection 2: Measurement Challenges}

\textbf{Objection:} Key constructs may be difficult to measure reliably in
practice.

\textbf{Response:} We recommend using validated scales and instruments where
available, and developing domain-specific proxies where necessary. Sensitivity
analysis should assess robustness to measurement uncertainty.

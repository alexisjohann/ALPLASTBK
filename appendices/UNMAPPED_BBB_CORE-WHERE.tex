\chapter{CORE-WHERE: Parameter Estimation and Sources}
\label{app:core-where}

\begin{abstract}
This appendix answers the fifth fundamental question:
\textit{``Where do parameter values come from?''}
We establish a four-tier estimation hierarchy:
(1) Literature-validated meta-analytic values,
(2) LLMMC (LLM Monte Carlo) informed priors,
(3) Empirical calibration from primary data,
(4) Expert elicitation with structured protocols.
Quality criteria ensure transparency about uncertainty and provenance.
\end{abstract}

\begin{tcolorbox}[colback=blue!5!white, colframe=blue!75!black,
    title=Quick Reference: CORE-WHERE]

\textbf{Core Question:} Where do parameter values come from?

\textbf{Estimation Hierarchy:}
\begin{enumerate}[nosep]
\item \textbf{Literature}: Meta-analyses, replicated experiments
\item \textbf{LLMMC}: LLM-informed priors with uncertainty
\item \textbf{Empirical}: Primary data collection
\item \textbf{Expert}: Structured elicitation

\section{Axiom System}
\label{sec:where-axioms}

\begin{tcolorbox}[colback=gray!5!white, colframe=gray!75!black,
    title=Axiom UNMAPPED_PAR-1: Provenance Requirement]
\textbf{Statement:} Every parameter must have documented provenance.
\begin{equation}
\forall \theta: \exists \text{source}(\theta) \in \{\text{LIT}, \text{LLMMC}, \text{EMP}, \text{EXP}\}
\end{equation}
\end{tcolorbox}

\begin{tcolorbox}[colback=gray!5!white, colframe=gray!75!black,
    title=Axiom UNMAPPED_PAR-2: Uncertainty Quantification]
\textbf{Statement:} Every parameter must include uncertainty bounds.
\begin{equation}
\theta = \hat{\theta} \pm \sigma_{\theta} \quad \text{or} \quad \theta \in [\theta_{\min}, \theta_{\max}]
\end{equation}
\end{tcolorbox}

\begin{tcolorbox}[colback=gray!5!white, colframe=gray!75!black,
    title=Axiom UNMAPPED_PAR-3: Context Indexing]
\textbf{Statement:} Parameters must be indexed by context.
\begin{equation}
\theta = \theta(\Psi_{\text{MACRO}}, \Psi_{\text{MESO}}, \Psi_{\text{MICRO}})
\end{equation}
\end{tcolorbox}

\begin{tcolorbox}[colback=red!5!white, colframe=red!75!black,
    title=Axiom UNMAPPED_PAR-4: Referential Integrity]
\textbf{Statement:} Every parameter used in a model must reference a canonical entry in the Parameter Registry.

\begin{equation}
\forall \theta \in \mathcal{M}: \exists \text{PAR-XXX} \in \mathcal{R} \text{ s.t. } \theta \mapsto \text{PAR-XXX}
\end{equation}

where $\mathcal{M}$ is the Model Registry and $\mathcal{R}$ is the Parameter Registry.

\textbf{Implementation:}
\begin{itemize}[nosep]
\item \texttt{model-registry.yaml}: Each parameter value must include \texttt{parameter\_ref: PAR-XXX}
\item \texttt{parameter-registry.yaml}: Single Source of Truth for all parameter values
\item Complementarity parameters ($\gamma_{ij}$) must reference \texttt{PAR-COMP-XXX}
\end{itemize}

\textbf{Rationale:} This ensures that parameter values are never ``invented'' in models but always trace back to documented, empirically-grounded sources with uncertainty quantification.
\end{tcolorbox}

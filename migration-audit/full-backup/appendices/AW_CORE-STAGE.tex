\chapter{CORE-STAGE: Behavioral Change Journey (BCJ) Phases}
\label{app:core-stage}

\begin{abstract}
This appendix answers the eighth fundamental question:
\textit{``Where is the individual in their behavioral change journey?''}
We formalize the five-phase BCJ model: (1) Unaware, (2) Aware, (3) Willing,
(4) Acting, (5) Maintaining. Each phase has characteristic barriers and
appropriate intervention types. Stage-mismatched interventions fail.
\end{abstract}

\begin{tcolorbox}[colback=blue!5!white, colframe=blue!75!black,
    title=Quick Reference: CORE-STAGE]

\textbf{Core Question:} Where is the individual in their change journey?

\textbf{The Five BCJ Phases:}
\begin{equation}
\text{Unaware} \xrightarrow{A>0} \text{Aware} \xrightarrow{W>0} \text{Willing} \xrightarrow{\text{act}} \text{Acting} \xrightarrow{\text{persist}} \text{Maintaining}
\end{equation}

\textbf{Phase Definitions:}
\begin{itemize}[nosep]
\item $\phi_0$: \textbf{Unaware} --- Does not know options/consequences exist
\item $\phi_1$: \textbf{Aware} --- Knows but not motivated to act
\item $\phi_2$: \textbf{Willing} --- Intends to act but hasn't
\item $\phi_3$: \textbf{Acting} --- Has begun but not habituated
\item $\phi_4$: \textbf{Maintaining} --- Behavior is stable/automatic
\end{itemize}

\textbf{Cross-References:}
AU (CORE-AWARE), AV (CORE-READY), IE (CORE-EIT), Chapter 18, CP (DOMAIN-CPS: Solution Phases)

\end{tcolorbox}

\section{The Fundamental Question}
\label{sec:stage-fundamental}

\begin{tcolorbox}[colback=red!5!white, colframe=red!75!black,
    title=The Fundamental Question]
\textbf{At what stage of the behavioral change journey is the individual?
What intervention types are appropriate for each stage?}
\end{tcolorbox}

The same intervention has radically different effects depending on stage:
\begin{itemize}
\item Information campaigns work for Unaware → Aware
\item Information campaigns fail for Willing → Acting (already aware!)
\item Commitment devices work for Willing → Acting
\item Commitment devices fail for Unaware (not ready to commit)
\end{itemize}

\subsection{Phase Transitions}

\begin{equation}
\phi_0 \xrightarrow[\text{intervention: info}]{A > \tau_A} \phi_1
\xrightarrow[\text{intervention: motivate}]{W > \tau_W} \phi_2
\xrightarrow[\text{intervention: commit}]{\text{first action}} \phi_3
\xrightarrow[\text{intervention: sustain}]{\text{habit formation}} \phi_4
\end{equation}

\section{Axiom System}
\label{sec:stage-axioms}

\begin{tcolorbox}[colback=gray!5!white, colframe=gray!75!black,
    title=Axiom BCJ-1: Sequential Progression]
\textbf{Statement:}
\begin{equation}
\phi_i \to \phi_j \text{ requires } j = i + 1 \quad \text{(no skipping phases)}
\end{equation}

\textbf{Interpretation:} Individuals must progress through phases sequentially.
One cannot go from Unaware directly to Acting.
\end{tcolorbox}

\begin{tcolorbox}[colback=gray!5!white, colframe=gray!75!black,
    title=Axiom BCJ-2: Regression Possibility]
\textbf{Statement:}
\begin{equation}
\phi_j \to \phi_i \text{ possible for } i < j \quad \text{(backsliding)}
\end{equation}

\textbf{Interpretation:} Regression to earlier phases is possible. Maintaining
($\phi_4$) can regress to Acting ($\phi_3$) or even Aware ($\phi_1$).
\end{tcolorbox}

\begin{tcolorbox}[colback=gray!5!white, colframe=gray!75!black,
    title=Axiom BCJ-3: Stage-Intervention Match]
\textbf{Statement:}
\begin{equation}
\text{Effectiveness}(I, \phi) \propto \alpha_{I,\phi} \quad \text{(affinity determines success)}
\end{equation}

\textbf{Interpretation:} Intervention effectiveness depends on stage-type affinity.
Mismatched interventions underperform regardless of implementation quality.
\end{tcolorbox}

\begin{tcolorbox}[colback=gray!5!white, colframe=gray!75!black,
    title=Axiom BCJ-4: Transition Thresholds]
\textbf{Statement:}
\begin{equation}
\phi_i \to \phi_{i+1} \text{ iff } \theta_{\text{transition}} > \tau_i
\end{equation}

\textbf{Interpretation:} Each transition has a threshold that must be exceeded.
Thresholds are context-dependent: $\tau_i = \tau_i(\Psi)$.
\end{tcolorbox}

\section{Stage Diagnosis}
\label{sec:stage-diagnosis}

\begin{equation}
\phi = \begin{cases}
\phi_0 & \text{if } A < \tau_A \\
\phi_1 & \text{if } A \geq \tau_A \text{ and } W < \tau_W \\
\phi_2 & \text{if } A \geq \tau_A, W \geq \tau_W, \text{ no action yet} \\
\phi_3 & \text{if acting, } t_{\text{since start}} < t_{\text{habit}} \\
\phi_4 & \text{if acting, } t_{\text{since start}} \geq t_{\text{habit}}
\end{cases}
\end{equation}

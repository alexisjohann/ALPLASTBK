\section{Abstract}
% =============================================================================

Raffaella Sadun's research program, primarily in collaboration with Nick Bloom
and John Van Reenen, has established \textbf{management practices} as a
measurable, consequential determinant of firm performance. Using the World
Management Survey (WMS) methodology, this work provides unprecedented
empirical evidence on how organizational practices---monitoring, targets,
incentives---create firm-level value.

This appendix integrates 29 papers spanning 2011--2025, organized around six
core axioms that connect Sadun's empirical findings to EBF's
complementarity-context framework.

\section{Fundamental Question}
% =============================================================================

\begin{tcolorbox}[colback=red!5!white, colframe=red!75!black,
    title=The Sadun Question]
\centering\Large
\textbf{Why do management practices vary so much across firms and countries,
and why does this variation matter for performance?}
\end{tcolorbox}

Sadun's answer: Management practices are \textit{technologies}---they can be
measured, transferred, and improved. Variation persists due to information
frictions, ownership structures (especially family firms), and institutional
context. The performance implications are large: a one-standard-deviation
improvement in WMS score corresponds to roughly 10\% higher productivity.

\section{The Six Sadun Axioms}
% =============================================================================

\subsection{Axiom RS-1: Management as Technology}

\begin{tcolorbox}[colback=blue!5!white, colframe=blue!75!black]
\textbf{Axiom RS-1 (Management Technology):}
\textit{Management practices constitute a measurable technology that explains
significant variation in firm productivity, comparable in magnitude to
R\&D and IT investments.}
\end{tcolorbox}

\textbf{Key Papers:}
\begin{itemize}[nosep]
    \item Bloom, Sadun \& Van Reenen (2016): ``Management as a Technology?''
    \item Bloom, Sadun \& Van Reenen (2012): ``Americans Do IT Better''
    \item Bloom et al. (2013): ``Does Management Matter? Evidence from India''
\end{itemize}

\textbf{EBF Integration:}
Management practices enter EBF as organizational-level parameters that
modulate the complementarity structure:
\begin{equation}
\gamma^{org}_{ij}(WMS) = \gamma^{base}_{ij} \cdot f(WMS_{score})
\end{equation}

Higher management quality increases the synergies between organizational
activities ($\gamma > 0$ becomes larger).

\subsection{Axiom RS-2: Complementarity in Practices}

\begin{tcolorbox}[colback=blue!5!white, colframe=blue!75!black]
\textbf{Axiom RS-2 (Practice Complementarity):}
\textit{Management practices are complementary: monitoring, targets, and
incentives work together as a system, not as isolated interventions.
Partial adoption yields limited returns.}
\end{tcolorbox}

\textbf{Key Papers:}
\begin{itemize}[nosep]
    \item Bloom, Sadun \& Van Reenen (2012): ``The Organization of Firms''
    \item Bloom et al. (2014): ``IT and Communication Technology Effects''
\end{itemize}

\textbf{EBF Integration:}
This directly supports CORE-HOW (B):
\begin{equation}
U^{management} = \sum_i u_i(p_i) + \sum_{i<j} \gamma_{ij} \cdot p_i \cdot p_j
\end{equation}
where $p_i \in \{monitoring, targets, incentives, ...\}$ and $\gamma_{ij} > 0$.

\subsection{Axiom RS-3: Decentralization and Context}

\begin{tcolorbox}[colback=blue!5!white, colframe=blue!75!black]
\textbf{Axiom RS-3 (Optimal Decentralization):}
\textit{Optimal decentralization depends on context: firms facing more
uncertainty, operating in high-trust environments, or with more educated
workers optimally delegate more decisions.}
\end{tcolorbox}

\textbf{Key Papers:}
\begin{itemize}[nosep]
    \item Bloom, Sadun \& Van Reenen (2012): ``Organization of Firms''
    \item Aghion, Bloom, Sadun \& Van Reenen (2021): ``Decentralization and Shocks''
\end{itemize}

\textbf{EBF Integration:}
Decentralization responds to context $\Psi$:
\begin{equation}
Decentralization^* = f(\Psi_{uncertainty}, \Psi_{trust}, \Psi_{education})
\end{equation}

This connects to CORE-WHEN (V): organizational structure adapts to context.

\subsection{Axiom RS-4: Family Firm Effects}

\begin{tcolorbox}[colback=blue!5!white, colframe=blue!75!black]
\textbf{Axiom RS-4 (Family Governance):}
\textit{Family ownership affects management quality through succession choices:
primogeniture (eldest son succession) significantly reduces management scores,
while professional CEO appointment preserves quality.}
\end{tcolorbox}

\textbf{Key Papers:}
\begin{itemize}[nosep]
    \item Bennedsen \& Sadun (2016): ``Family Business Map''
    \item Bennedsen \& Sadun (2017): ``Corporate Governance and Family Firms''
    \item Bennedsen \& Sadun (2021): ``Death of the Founder''
\end{itemize}

\textbf{EBF Integration:}
Family ownership creates identity-based constraints on organizational utility:
\begin{equation}
U^{family\_firm} = U^{economic} + \lambda_{family} \cdot U^{IDN}_{family\_control}
\end{equation}

The $\lambda_{family}$ weight on family identity may reduce economic efficiency.

\subsection{Axiom RS-5: CEO Behavior Matters}

\begin{tcolorbox}[colback=blue!5!white, colframe=blue!75!black]
\textbf{Axiom RS-5 (Leadership Time):}
\textit{How CEOs allocate their time---across meetings, alone time,
internal vs. external focus---significantly predicts firm performance.
Leader behavior is measurable and consequential.}
\end{tcolorbox}

\textbf{Key Papers:}
\begin{itemize}[nosep]
    \item Bandiera, Sadun et al. (2020): ``CEO Behavior and Firm Performance''
    \item Bandiera, Sadun \& Prat (2011): ``What Do CEOs Do?''
\end{itemize}

\textbf{EBF Integration:}
CEO time allocation affects organizational awareness $A^{org}$:
\begin{equation}
A^{org}_{issue} = f(CEO\_attention_{issue}, Communication_{structure})
\end{equation}

What leaders focus on becomes salient for the organization.

\subsection{Axiom RS-6: Work Organization Evolution}

\begin{tcolorbox}[colback=blue!5!white, colframe=blue!75!black]
\textbf{Axiom RS-6 (Hybrid Work):}
\textit{Remote and hybrid work arrangements affect productivity, retention,
and well-being through multiple channels. Optimal work organization
balances coordination needs against flexibility benefits.}
\end{tcolorbox}

\textbf{Key Papers:}
\begin{itemize}[nosep]
    \item Bloom, Sadun et al. (2024): ``Hybrid Working Improves Retention'' (Nature)
    \item Barrero, Bloom, Sadun \& Davis (2021): ``How Working from Home Works Out''
\end{itemize}

\textbf{EBF Integration:}
Work arrangement affects multiple utility dimensions:
\begin{equation}
\Delta U^{hybrid} = \Delta U^{Financial} + \Delta U^{Emotional} + \Delta U^{Social}
\end{equation}

Hybrid work may increase emotional utility (flexibility) while affecting
social utility (connection) with ambiguous net effects.

\section{World Management Survey: Empirical Foundation}
% =============================================================================

\begin{tcolorbox}[colback=green!5!white, colframe=green!75!black,
    title=WMS Methodology]

The World Management Survey provides the empirical backbone for Sadun's research:

\textbf{Coverage:}
\begin{itemize}[nosep]
    \item 35,000+ firms across 35 countries
    \item Manufacturing, retail, healthcare, education sectors
    \item 2004--present (ongoing data collection)
\end{itemize}

\textbf{Measurement:}
\begin{itemize}[nosep]
    \item 18 questions on management practices
    \item Scored 1--5 by trained interviewers
    \item Double-blind methodology (interviewer doesn't know performance)
\end{itemize}

\textbf{Three Dimensions:}
\begin{enumerate}[nosep]
    \item \textbf{Monitoring}: Tracking performance, using data
    \item \textbf{Targets}: Setting stretch goals, balancing short/long-term
    \item \textbf{Incentives}: Rewarding performance, addressing underperformance
\end{enumerate}

\end{tcolorbox}

\section{Worked Example: Healthcare Management}
% =============================================================================

\begin{tcolorbox}[colback=green!5!white, colframe=green!75!black,
    title=Example: Hospital Management and Patient Outcomes]

\textbf{Context:} NHS hospitals vary significantly in management quality.

\textbf{Sadun Finding (2015, 2016):}
\begin{itemize}
    \item 1 SD increase in hospital WMS score $\rightarrow$ significant reduction
    in mortality for acute conditions (AMI, hip fracture)
    \item Board composition and governance structures predict management quality
\end{itemize}

\textbf{EBF Application:}
\begin{equation}
U^{patient}_{health} = f(Clinical\_quality, \underbrace{Management\_quality}_{WMS})
\end{equation}

Management practices (monitoring, targets, incentives) create complementarities
that improve clinical outcomes:
\begin{equation}
\gamma_{clinical, management} > 0
\end{equation}

\textbf{Policy Implication:}
Healthcare policy should target management practices alongside clinical
interventions. The EBF framework predicts larger effects when both are
improved together (complementarity).

\end{tcolorbox}

\section{Critical Foundations}
% =============================================================================

\begin{enumerate}
    \item \textbf{Methodological Rigor:} WMS provides high-quality, cross-country
    comparable data on management (Evidence Tier 1).

    \item \textbf{Causal Identification:} India RCT (Bloom et al. 2013) provides
    clean causal evidence that management practices cause productivity gains.

    \item \textbf{Complementarity Evidence:} Empirical support for $\gamma > 0$
    between practices, IT, and decentralization.

    \item \textbf{External Validity:} Results replicate across manufacturing,
    healthcare, education, and retail sectors.

    \item \textbf{Policy Relevance:} Direct implications for organizational
    design, healthcare policy, and development economics.
\end{enumerate}

\section{Open Issues}
% =============================================================================

\begin{itemize}
    \item \textbf{Adoption Barriers:} Why don't poorly-managed firms adopt
    better practices? Information, incentives, or capabilities?

    \item \textbf{Cultural Limits:} Do management practices transfer across
    cultural contexts? WMS is Western-developed.

    \item \textbf{Worker Welfare:} Do ``good'' management practices benefit
    workers or only shareholders?

    \item \textbf{AI and Management:} How will AI change optimal management
    practices? Early research ongoing.
\end{itemize}

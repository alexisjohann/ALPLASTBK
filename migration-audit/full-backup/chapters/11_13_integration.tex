\subsubsection{The Bias Block Coupling}
%------------------------------------------------------------------------------

Awareness is coupled with a Bias Block that captures systematic distortions:

\begin{equation}
Bias_{total} = Bias_{conf} + Bias_{surv} + Bias_{fail} + \kappa(Bias_{conf} \times Bias_{surv} \times Bias_{fail})
\end{equation}

\begin{table}[htbp]
\centering
\caption{Bias Block Components}
\label{tab:bias-block}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{@{}lll@{}}
\toprule
\textbf{Component} & \textbf{Symbol} & \textbf{Effect on Awareness} \\
\midrule
Overconfidence & $Bias_{conf}$ & Inflates $A$ for success, deflates for failure \\
Survivorship & $Bias_{surv}$ & $A_{success} >> A_{failure}$ \\
Failure Gap & $Bias_{fail}$ & $\Delta A_F$ from visibility costs \\
Interaction & $\kappa$ & Amplification when multiple biases present \\
\bottomrule
\end{tabular}
\end{table}

\paragraph{Effective Awareness with Bias.}
\begin{equation}
A_{eff}(t^*) = A_{base}(t^*) \times (1 - Bias_{total})
\end{equation}

\subsubsection{Meta-Constraints}
%------------------------------------------------------------------------------

Four meta-constraints govern awareness validity:

\begin{enumerate}
\item[\textbf{C1:}] Awareness is only valid when relevant context ($C_{rel}$) is present
\item[\textbf{C2:}] Awareness cannot be simultaneously maximal in SYS 0 and SYS 2 (Hyper-Dissonance)
\item[\textbf{C3:}] KNU is always awareness-mandatory; instant KNU is impossible
\item[\textbf{C4:}] Awareness drops when complexity or uncertainty exceeds threshold
\end{enumerate}

\paragraph{Validation Rule.}
\begin{equation}
\sum |\Delta A_{SYS_i}| \leq \theta_{meta} \wedge C1-C4 \text{ fulfilled} \Rightarrow \text{Awareness} = \text{VALID}
\end{equation}

\begin{itemize}[nosep]
    \item Read chapters 1--24 in order
    \item Full understanding of framework evolution
    \item Time: Comprehensive study
\end{itemize}

\textbf{Path 2: CORE First} (6 chapters + foundations)
\begin{itemize}[nosep]
    \item Start: Ch. 1 (Introduction)
    \item Then: Ch. 5, 9, 10, 11, 12, 13 (all TYPE A chapters)
    \item Appendices: B, V, C, AU, AV, AW (CORE appendices)
    \item Result: Complete theoretical foundation
\end{itemize}

\textbf{Path 3: Application Focus} (10 chapters)
\begin{itemize}[nosep]
    \item Prerequisites: Ch. 1, 10, 13 (minimum foundation)
    \item Then: Ch. 14, 17--24 (all TYPE C chapters)
    \item Result: Practical implementation knowledge
\end{itemize}

\textbf{Path 4: Intervention Design} (5 chapters)
\begin{itemize}[nosep]
    \item Prerequisites: Ch. 13 (BCJ), Ch. 14 (Segments)
    \item Core: Ch. 17 (Foundations), Ch. 18 (Journey), Ch. 19 (Segments)
    \item Advanced: Ch. 20 (Portfolios), Ch. 21 (Evolution)
    \item Result: Complete intervention design capability
\end{itemize}

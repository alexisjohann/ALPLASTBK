clusters. This section covers the first four foundational clusters.

%------------------------------------------------------------------------------
\subsubsection{Cluster 1: Fundamental Axioms (AWX-A1 to AWX-A10)}
%------------------------------------------------------------------------------

These axioms establish the basic properties of awareness.

\begin{table}[htbp]
\centering
\caption{Cluster 1: Fundamental Axioms}
\label{tab:cluster1-axioms}
\renewcommand{\arraystretch}{1.3}
\small
\begin{tabular}{@{}llp{7cm}@{}}
\toprule
\textbf{ID} & \textbf{Title} & \textbf{Core Statement} \\
\midrule
AWX-A1 & Transformation Condition & $U_{eff} = U_{pot} \times A$; awareness is necessary for behavior \\
AWX-A2 & Moment of Truth & Awareness operates only at $t^*$, not before or after \\
AWX-A3 & Positive \& Negative Impact & Awareness encompasses both gains and pains \\
AWX-A4 & Instant Exceptions & INU/IDN can be instant; KNU never \\
AWX-A5 & General vs. Situational & Only situational awareness at $t^*$ is behaviorally effective \\
AWX-A6 & Individual Differences & Awareness varies between individuals \\
AWX-A7 & Context Dependence & Awareness is completely context-bound: $A = f(C)$ \\
AWX-A8 & Capacity Limitation & $\sum A_X \leq Capacity_{limit}$ \\
AWX-A9 & Salience \& Priming & Awareness requires salience or priming \\
AWX-A10 & Habituation & Habitualized behavior requires no awareness \\
\bottomrule
\end{tabular}
\end{table}

\paragraph{Key Sub-axioms.}
\begin{itemize}
\item \textbf{AWX-A1a (Community Relevance):} Awareness includes effects on relevant community
\item \textbf{AWX-A1b (External Effects):} Awareness is only effective when external effects are recognized
\item \textbf{AWX-A2a (Situational Activation):} Only context-triggered awareness is effective
\item \textbf{AWX-A2b (Learning Effect):} General awareness builds through repeated situational activation
\item \textbf{AWX-A9a (Hyper-Awareness):} Awareness can be excessively heightened
\item \textbf{AWX-A9b (Awareness Blockade):} Awareness can be blocked despite relevance
\item \textbf{AWX-A10a (Decay Effect):} Awareness decays with habituation
\end{itemize}

\paragraph{The Failure Awareness Gap (AWX-A47a).}
This axiom deserves special attention:

\begin{equation}
\Delta A_F = f(VisibilityCost, PowerDistance, -Transparency, -PsychSafety, FeedbackLag, NormShame)
\end{equation}

\begin{equation}
A_{eff}(t^*) = A_{base}(t^*) \times (1 - \Delta A_F)
\end{equation}

\paragraph{Critical Threshold.}
\begin{equation}
\Delta A_F > 0.25 \quad \Rightarrow \quad \text{Perceptual Lock-In (Watchdog Alert)}
\end{equation}

\subsubsection{Summary: Clusters 1--4}
%------------------------------------------------------------------------------

\begin{tcolorbox}[colback=gray!5!white, colframe=gray!75!black, title=Axiom Clusters 1--4 Summary]

\textbf{Cluster 1 (Fundamental):} 10 axioms establishing core properties
\begin{itemize}
\item Awareness as transformation condition
\item Moment of Truth ($t^*$)
\item Instant exceptions (INU, IDN yes; KNU never)
\item Habituation removes awareness requirement
\end{itemize}

\textbf{Cluster 2 (Determinants):} 20 axioms specifying what shapes $A$
\begin{itemize}
\item Cognitive: attention, load, evidence, availability
\item Emotional: valence, fear, hope, resonance
\item Social: norms, peers, networks
\item Institutional: legitimacy, trust, rules
\item Cultural/Biological: narratives, symbols, energy
\end{itemize}

\textbf{Cluster 3 (Utility-Specific):} 15 axioms for category differences
\begin{itemize}
\item Hierarchy: $A_{INU} > A_{IDN} > A_{KNU}$
\item Instant rules for each category
\item Cross-category interactions
\end{itemize}

\textbf{Cluster 4 (Complexity/Uncertainty):} 10 axioms for difficult situations
\begin{itemize}
\item Complexity reduces awareness
\item Failure Awareness Gap
\item Multiple uncertainty types compound
\end{itemize}

\end{tcolorbox}

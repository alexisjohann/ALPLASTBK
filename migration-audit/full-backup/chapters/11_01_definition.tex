Awareness requires two complementary definitions: a \emph{content definition}
(what awareness is about) and a \emph{functional definition} (how awareness operates).

%------------------------------------------------------------------------------
\subsubsection{The Content Definition}
%------------------------------------------------------------------------------

\begin{definition}[Awareness -- Content]
\label{def:awareness-content}
\textbf{Awareness} is the consciousness of the positive and negative effects 
on one's own utility level (INU, KNU, IDN) and on the utility level (INU, KNU, IDN) 
of a relevant community, that a person has at the moment of decision or 
exhibited behavior.
\end{definition}

\paragraph{In German:}
\textit{Awareness ist das Bewusstsein über die positiven und negativen Effekte 
für das eigene Nutzenlevel (INU, KNU, IDN) und das Nutzenlevel (INU, KNU, IDN) 
einer relevanten Community, die ein Mensch im Moment der Entscheidung / 
des gezeigten Verhaltens hat.}

%------------------------------------------------------------------------------
\subsubsection{The Functional Definition}
%------------------------------------------------------------------------------

\begin{definition}[Awareness -- Function]
\label{def:awareness-function}
\textbf{Awareness} is the necessary condition for transforming potential utility
into effective utility at the Moment of Truth ($t^*$).
\end{definition}

\paragraph{The Core Transformation Formula.}
\begin{equation}
\boxed{U_{eff,X}(t^*) = U_{pot,X}(t^*) \times A_X(t^*)}
\label{eq:awareness-transformation}
\end{equation}

where:
\begin{itemize}
\item $U_{pot,X}$: Potential utility of type $X \in \{INU, KNU, IDN\}$
\item $U_{eff,X}$: Effective utility (actually guides behavior)
\item $A_X(t^*)$: Awareness level at Moment of Truth
\item $A_X \in [0, 1]$: Continuous from no awareness to full awareness
\end{itemize}

\paragraph{Interpretation.}
Awareness functions as a \textbf{multiplicative filter}:
\begin{itemize}
\item $A = 0$: Utility exists but has zero behavioral impact
\item $A = 0.5$: Half of potential utility becomes effective
\item $A = 1$: Full potential utility becomes effective
\end{itemize}

Both are essential for understanding and applying the construct.

%------------------------------------------------------------------------------
\subsubsection{The Knowledge-Awareness Distinction}
%------------------------------------------------------------------------------

A critical distinction: \textbf{knowledge is not awareness}.

\begin{equation}
\text{Knowledge} \not\Rightarrow \text{Awareness}
\end{equation}

\begin{table}[htbp]
\centering
\caption{Knowledge vs. Awareness}
\label{tab:knowledge-vs-awareness}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{@{}lll@{}}
\toprule
\textbf{State} & \textbf{Description} & \textbf{Behavioral Impact} \\
\midrule
No knowledge & Person doesn't know the effect exists & None \\
Latent knowledge & Knows but not thinking about it now & None \\
\textbf{Conscious awareness} & \textbf{Actively thinking about it} & \textbf{Guides behavior} \\
\bottomrule
\end{tabular}
\end{table}

\paragraph{Example: Smoking.}
A smoker may \emph{know} that smoking causes cancer ($U^{INU}_{P,3}$ effect)
but not be \emph{aware} of this at the moment of lighting a cigarette.
Knowledge exists in memory; awareness exists in working memory at $t^*$.

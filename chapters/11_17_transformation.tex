}
\label{eq:meta-formula-complete}
\end{equation}

Where:
\begin{itemize}

    \item $\alpha_{d,s,j}$: Elasticity of determinant $d$ on component $(s,j)$
    \item $\Psi_k$: Eight context dimensions

\subsubsection{10-Step Computational Implementation}

The meta-formula translates into a 10-step algorithm:

\begin{enumerate}
    \item \textbf{Initialize} base awareness $A^{(0)}$ for 6 components 
          (self/comm × INU/KNU/IDN)
    
    \item \textbf{Apply} 20 determinants with component-specific elasticities:
    \begin{equation}
    A^{(1)}_{s,j} = A^{(0)}_{s,j} \cdot \prod_{d=1}^{20} D_d^{\alpha_{d,s,j}}
    \end{equation}
    
    \item \textbf{Differentiate} by utility type (instant rules):
    \begin{equation}
    A^{(2)}_{s,j} = \begin{cases}
    1 & \text{if } j = IDN \text{ and instant threat} \\
    0 & \text{if } j = KNU \text{ and instant} \\
    A^{(1)}_{s,j} & \text{otherwise}
    \end{cases}
    \end{equation}
    
    \item \textbf{Apply} K×U complexity-uncertainty penalty:
    \begin{equation}
    A^{(3)}_{s,j} = A^{(2)}_{s,j} \cdot (1 - K \times U)
    \end{equation}
    
    \item \textbf{Apply} temporal decay for future consequences:
    \begin{equation}
    A^{(4)}_{s,j}(t) = A^{(3)}_{s,j} \cdot e^{-\lambda_j t}
    \end{equation}
    
    \item \textbf{Check} psychological distortions:
    \begin{equation}
    A^{(5)}_{s,j} = A^{(4)}_{s,j} \cdot (1 - \sigma \cdot Threat) \cdot Split \cdot NormConflict
    \end{equation}
    
    \item \textbf{Modulate} by population segment:
    \begin{equation}
    A^{(6)}_{s,j} = A^{(5)}_{s,j} \cdot Segment_{modifier}
    \end{equation}
    
    \item \textbf{Aggregate} to collective level with cascades:
    \begin{equation}
    A^{(7)}_{collective} = f(A^{(6)}_{individual}, Network) \cdot Cascade
    \end{equation}
    
    \item \textbf{Apply} belief filter:
    \begin{equation}
    A^{(8)}_{s,j} = A^{(7)}_{s,j} \cdot B_{belief}
    \end{equation}
    where $B_{belief} = 1$ for consistent, $B = (1 - \lambda |U'_{IDN}|)$ for inconsistent
    
    \item \textbf{Modulate} by 8 Ψ-context dimensions:
    \begin{equation}
    A^{(final)}_{s,j} = A^{(8)}_{s,j} \cdot \prod_{k=1}^{8} (1 + \gamma_k \cdot \Psi_k)
    \end{equation}
\end{enumerate}

\subsubsection{The Effective Utility Calculation}

With the complete awareness vector, effective utility is computed:

\begin{equation}
U_{eff}(t^*) = \sum_{s \in \{self,comm\}} \sum_{j \in \{INU,KNU,IDN\}} A_{s,j}(t^*) \cdot U_{s,j,pot}
\label{eq:effective-utility-complete}
\end{equation}

The sign of $U_{eff}$ determines behavior:
\begin{itemize}
    \item $U_{eff} > 0$: Action is taken
    \item $U_{eff} < 0$: Action is avoided
    \item $U_{eff} \approx 0$: Indifference (random or habitual choice)
\end{itemize}
